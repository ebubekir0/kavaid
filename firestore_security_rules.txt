// Firebase Firestore Security Rules - Mevcut + Chat System
// Bu kuralları Firebase Console > Firestore Database > Rules bölümüne kopyalayın

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // MEVCUT KURALLAR - Kullanıcı dokümanları ve kaydedilen kelimeler
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;

      // Alt koleksiyon: saved_words
      match /saved_words/{docId} {
        allow read, write: if request.auth != null && request.auth.uid == userId;
      }
    }
    
    // YENİ CHAT SİSTEMİ - Genel sohbet odası
    match /chats/general_chat/messages/{messageId} {
      // Herkes mesajları okuyabilir (giriş yapmış kullanıcılar)
      allow read: if request.auth != null;
      
      // Sadece giriş yapmış kullanıcılar mesaj gönderebilir
      allow create: if request.auth != null 
        && request.auth.uid == request.resource.data.senderId
        && request.resource.data.keys().hasAll(['content', 'senderId', 'senderName', 'timestamp'])
        && request.resource.data.content is string
        && request.resource.data.content.size() > 0
        && request.resource.data.content.size() <= 500; // Max 500 karakter
      
      // Sadece kendi mesajlarını silebilir
      allow delete: if request.auth != null 
        && request.auth.uid == resource.data.senderId;
      
      // Mesaj düzenleme yasak
      allow update: if false;
    }
  }
}
